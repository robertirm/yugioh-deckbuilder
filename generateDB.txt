create table cards
(
    card_id           integer      not null,
    card_name         varchar(50)  not null
        constraint cards_card_name_key
            unique,
    card_type         varchar(10)  not null,
    card_subtype      varchar(50)  not null,
    monster_level     integer,
    monster_attribute varchar(10),
    monster_atk       integer,
    monster_def       integer,
    card_text         varchar(500) not null,
    id                serial
        constraint cards_pk
            primary key
);

alter table cards
    owner to postgres;

create unique index cards_id_uindex
    on cards (id);
